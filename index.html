import random # --- Equipos --- primera = ["La 23","Los Verdes","Sol de América","La Riveira","El Plan", "Sporting","NTV","Matadores","Garrablanca","La Riveira2", "Los Tigres","Deportivo América","River B","River C","Equipo15","Equipo16"] bnacional = ["Güemes","Capital","Persa FC","La Arboledad","Carrefourt", "Unión Infantil","Unión Hurlingham","Celestial","La Base","El Gol", "Sevilla","San Ciro","El Fortín","San José","Deportivo Motores","Deportivo Morón"] # Moral inicial moral_primera = {e:50 for e in primera} moral_bn = {e:50 for e in bnacional} # Tabla inicial tabla_primera = {e: {"puntos":0,"gf":0,"gc":0,"dif":0} for e in primera} tabla_bn = {e: {"puntos":0,"gf":0,"gc":0,"dif":0} for e in bnacional} # --- Generar partidos --- def generar_partidos(equipos, tu_equipo): equipos_disp = equipos.copy() random.shuffle(equipos_disp) partidos = [] while equipos_disp: e1 = equipos_disp.pop(0) if e1 == tu_equipo: e2 = random.choice([x for x in equipos_disp if x != tu_equipo]) equipos_disp.remove(e2) partidos.append((e1,e2,"TU")) else: if not equipos_disp: break e2 = equipos_disp.pop(0) partidos.append((e1,e2,"NORMAL")) return partidos # --- Simular partido --- def simular(e1,e2,tipo,moral,tabla): mod1 = random.randint(0,5) + moral[e1]//20 mod2 = random.randint(0,5) + moral[e2]//20 res = (mod1,mod2) # Tabla tabla[e1]["gf"] += mod1 tabla[e1]["gc"] += mod2 tabla[e2]["gf"] += mod2 tabla[e2]["gc"] += mod1 tabla[e1]["dif"] = tabla[e1]["gf"]-tabla[e1]["gc"] tabla[e2]["dif"] = tabla[e2]["gf"]-tabla[e2]["gc"] # Puntos y moral if mod1>mod2: tabla[e1]["puntos"]+=3 moral[e1]+=1 moral[e2]-=1 elif mod2>mod1: tabla[e2]["puntos"]+=3 moral[e2]+=1 moral[e1]-=1 else: tabla[e1]["puntos"]+=1 tabla[e2]["puntos"]+=1 # Noticias noticias=[] if tipo=="TU": noticias.append(f"⚽ Tu partido: {e1} {mod1}-{mod2} {e2}") if abs(mod1-mod2)>=3: noticias.append(f"🔥 Goleada: {e1} {mod1}-{mod2} {e2}") return res, noticias # --- Simular fecha completa --- def simular_fecha(tu_equipo_primera, tu_equipo_bn=None): print("\n=== Simulando Fecha Primera ===") partidos = generar_partidos(primera, tu_equipo_primera) noticias_fecha=[] for e1,e2,tipo in partidos: res, noticias = simular(e1,e2,tipo,moral_primera,tabla_primera) noticias_fecha.extend(noticias) print(f"{e1} {res[0]} - {res[1]} {e2}") print("\n=== Simulando Fecha B Nacional ===") if tu_equipo_bn: partidos_bn = generar_partidos(bnacional, tu_equipo_bn) for e1,e2,tipo in partidos_bn: res, noticias = simular(e1,e2,tipo,moral_bn,tabla_bn) noticias_fecha.extend(noticias) print(f"{e1} {res[0]} - {res[1]} {e2}") print("\n📰 Noticias importantes:") for n in noticias_fecha: print(n) # Mostrar tabla Primera print("\n📊 Tabla Primera División:") lider = max(tabla_primera.items(), key=lambda x:x[1]["puntos"])[0] for eq,data in tabla_primera.items(): icono=" 👑" if eq==lider else "" print(f"{eq}: {data['puntos']} pts GF:{data['gf']} GC:{data['gc']} DIF:{data['dif']}{icono}") # Mostrar tabla B Nacional print("\n📊 Tabla B Nacional:") lider_bn = max(tabla_bn.items(), key=lambda x:x[1]["puntos"])[0] for eq,data in tabla_bn.items(): icono=" 👑" if eq==lider_bn else "" print(f"{eq}: {data['puntos']} pts GF:{data['gf']} GC:{data['gc']} DIF:{data['dif']}{icono}") # --- Simular promoción automática --- def promocion(): # 14° Primera vs 3° B Nacional equipos_primera_ord = sorted(tabla_primera.items(), key=lambda x:x[1]["puntos"]) equipos_bn_ord = sorted(tabla_bn.items(), key=lambda x:x[1]["puntos"], reverse=True) p1 = equipos_primera_ord[2][0] # 14° (índice 2 desde el final) b3 = equipos_bn_ord[2][0] # 3° B Nacional print(f"\n⚔️ Promoción: {p1} vs {b3} (ida y vuelta)") # Ida ida = random.randint(0,5), random.randint(0,5) # Vuelta vuelta = random.randint(0,5), random.randint(0,5) total_primera = ida[0]+vuelta[1] total_bn = ida[1]+vuelta[0] print(f"Ida: {p1} {ida[0]}-{ida[1]} {b3}") print(f"Vuelta: {p1} {vuelta[0]}-{vuelta[1]} {b3}") if total_primera > total_bn: print(f"✅ {p1} se mantiene en Primera División") elif total_bn > total_primera: print(f"✅ {b3} asciende a Primera División") else: # Penales penales_primera = random.randint(0,5) penales_bn = random.randint(0,5) ganador = p1 if penales_primera>penales_bn else b3 print(f"⚡ Empate global, penales: {penales_primera}-{penales_bn} → {ganador} gana la promoción") # --- EJEMPLO DE USO --- tu_equipo_primera = "La 23" tu_equipo_bn = "Güemes" simular_fecha(tu_equipo_primera, tu_equipo_bn) promocion()
